📚 Публичный API Документация

Используйте эти эндпоинты для получения данных о прокси на вашем сайте. Авторизация не требуется.

🌐 Базовый URL

{{ base_url }}

📡 Эндпоинты

МетодЭндпоинтОписаниеАвторизация
GET /api/public/proxies Все прокси сервера (MTProto + SOCKS5 + HTTP) ❌ Не требуется
GET /api/public/mtproto Только MTProto прокси (массив ссылок) ❌ Не требуется
GET /api/public/docs Эта страница документации ❌ Не требуется

📦 Ответ /api/public/proxies

Возвращает все прокси сервера: MTProto, SOCKS5 и HTTP.

{
  "updated": "2026-04-05T12:00:00",
  "server_ip": "{{ server_ip }}",
  "mtproto": [
    {
      "label": "main",
      "port": 443,
      "domain": "cloudflare.com",
      "secret": "ee...",
      "link": "tg://proxy?server={{ server_ip }}&port=443&secret=ee..."
    }
  ],
  "socks5": {
    "enabled": true,
    "port": 1080,
    "link": "socks5://{{ server_ip }}:1080"
  },
  "http": {
    "enabled": true,
    "port": 3128,
    "link": "http://{{ server_ip }}:3128"
  }
}

📦 Ответ /api/public/mtproto

Возвращает только MTProto прокси — массив ссылок для подключения.

{
  "proxies": [
    {
      "label": "main",
      "port": 443,
      "domain": "cloudflare.com",
      "secret": "ee...",
      "link": "tg://proxy?server={{ server_ip }}&port=443&secret=ee..."
    }
  ],
  "updated": "2026-04-05T12:00:00"
}

💻 Примеры использования

JavaScript (fetch)

// Получить все прокси
const res = await fetch('{{ base_url }}/api/public/proxies');
const data = await res.json();

// MTProto прокси
data.mtproto.forEach(proxy => {
  console.log(proxy.link); // tg://proxy?server=...
});

// SOCKS5
if (data.socks5.enabled) {
  console.log(data.socks5.link);
}

// HTTP
if (data.http.enabled) {
  console.log(data.http.link);
}

Python

import requests

res = requests.get('{{ base_url }}/api/public/proxies')
data = res.json()

for proxy in data['mtproto']:
    print(proxy['link'])

cURL

curl {{ base_url }}/api/public/proxies

PHP

<?php
$data = json_decode(file_get_contents('{{ base_url }}/api/public/proxies'), true);
foreach ($data['mtproto'] as $proxy) {
    echo $proxy['link'] . "\n";
}
?>
